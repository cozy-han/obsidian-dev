# 프로젝트 구조

```
src/
├── pages/              # Next.js 페이지 라우팅
│   ├── api/            # API Routes (BFF)
│   ├── booking/        # 예약 플로우
│   ├── contents/       # 콘텐츠 페이지
│   ├── corporation/    # 법인 회원
│   ├── customer/       # 인증 (로그인/회원가입)
│   │   ├── guest/      # 비회원 예약
│   │   └── member/     # 회원 (가입/비밀번호 재설정)
│   ├── flightStatus/   # 운항 현황
│   ├── itinerary/      # 여정 관리
│   │   ├── myflight/   # 내 예약 조회
│   │   └── webcheckin/ # 웹 체크인
│   ├── mypage/         # 마이페이지
│   └── schedule/       # 운항 스케줄
│
├── components/         # 컴포넌트
│   ├── booking/        # 예약 관련 컴포넌트
│   ├── common/         # 공통 컴포넌트 (Layout, Header, Footer, Custom MUI 등)
│   ├── customer/       # 인증 관련 컴포넌트
│   ├── flightStatus/   # 운항현황 컴포넌트
│   ├── itinerary/      # 여정관리 컴포넌트
│   ├── main/           # 메인페이지 컴포넌트
│   ├── mypage/         # 마이페이지 컴포넌트
│   ├── navigation/     # 네비게이션
│   └── webcheckin/     # 웹체크인 컴포넌트
│
├── service/            # API 호출 레이어 (React Query + Zustand)
│   ├── booking.ts      # 항공편 검색/예약
│   ├── customer.ts     # 인증/회원관리 (40+개 엔드포인트)
│   ├── itinerary.ts    # 여정/예약 관리 (30+개 엔드포인트)
│   ├── main.ts         # 메인페이지 콘텐츠
│   ├── schedule.ts     # 운항 스케줄
│   ├── webcheckin.ts   # 웹 체크인
│   └── common/         # 공통 서비스
│
├── store/              # Zustand 상태 저장소
│   ├── useLoginDataStore.ts    # 로그인 데이터
│   ├── useCountStore.ts        # 카운터
│   ├── useGuestBookingStore.ts # 비회원 예약 정보
│   ├── useAuthNumberStore.ts   # 인증번호 상태
│   ├── itineraryStore.ts       # 여정 정보
│   ├── webCheckInStore.ts      # 웹체크인 정보
│   └── flightInquiry/          # 운항조회 상태
│       └── flightStatusStore.ts
│
├── types/              # TypeScript 타입 정의
│   ├── booking/        # 예약 (passenger, payment, avail, ancillary, seat)
│   ├── common/         # 공통 타입
│   ├── customer/       # 회원 타입
│   ├── flightInquiry/  # 운항조회 타입
│   ├── itinerary/      # 여정 타입
│   ├── main/           # 메인페이지 타입
│   ├── schedule/       # 스케줄 타입
│   ├── seat/           # 좌석 타입
│   └── webcheckin/     # 웹체크인 타입
│
├── config/             # 설정
│   ├── constants.ts    # 환경변수/상수
│   └── i18n.ts         # 다국어 설정
│
├── locales/            # 번역 파일
│   ├── ko/             # 한국어 (12개 네임스페이스)
│   └── en/             # 영어
│
├── styles/             # 스타일
│   ├── theme.ts        # MUI 테마
│   ├── reset.css       # CSS 리셋
│   ├── common.css      # 공통 스타일
│   └── fonts/          # 아이콘폰트 (icomoon)
│
├── utility/            # 유틸리티
│   ├── index.ts        # 포맷팅, 검증, 쿠키 등 (530+ lines)
│   ├── axiosInterceptor.ts  # HTTP 클라이언트
│   ├── payment.ts      # 결제 유틸
│   ├── nativeTokenBridge.ts # RN WebView 토큰 브릿지
│   ├── createEmotionCache.ts # Emotion SSR
│   └── countries.ts    # 국가 목록
│
├── staticData/         # 정적 데이터
│   ├── flight/stepList.ts  # 예약 단계 정의
│   └── join/stepList.ts    # 회원가입 단계 정의
│
└── middleware.ts       # Edge 미들웨어 (인증 라우트 보호)
```

## 공통 컴포넌트 (components/common/)

| 컴포넌트 | 설명 |
|---------|------|
| Layout.tsx | 공통 레이아웃 |
| header/ | 헤더 |
| footer/ | MainFooter, SubFooter |
| custom/ | MUI 커스텀 래퍼들 (accordion, alert, autocomplete, box, button, checkbox, container, dialog, formControl, grid, iconButton, modal, select, slideDialog, tabPanel, tabs, textField) |
| datepicker/ | 날짜선택기 (Single/Range, Mobile/PC) |
| infoForm/ | 정보입력폼 (Name, Birthday, Gender, Country, Phone) |
| seat/ | 좌석 선택 공통 |
| step/ | 단계 인디케이터 |
| terms/ | 약관 동의 |
| complete/ | 완료 페이지 |
| flightInquiry/ | 항공편 조회 |
| authNumber/ | 인증번호 입력 |
